CMAKE_MINIMUM_REQUIRED(VERSION 2.8.8)

FIND_PACKAGE(Aspect 1.5.0 QUIET HINTS ${Aspect_DIR} ../ ../../ $ENV{ASPECT_DIR})

IF (NOT Aspect_FOUND)
  MESSAGE(FATAL_ERROR "\n"
	"Could not find a valid ASPECT build/installation directory. "
	"Please specify the directory where you are building ASPECT by passing\n"
	"   -D Aspect_DIR=<path to ASPECT>\n"
	"to cmake or by setting the environment variable ASPECT_DIR in your shell "
	"before calling cmake. See the section 'How to write a plugin' in the "
        "manual for more information.")
ENDIF ()

IF (NOT ASPECT_SOURCE_DIR)
  MESSAGE(FATAL_ERROR "\n"
    "Please specify the path to the source directory of ASPECT by passing\n"
	"-DASPECT_SOURCE_DIR=<path to ASPECT source>\n"
	"to cmake. This is required since solkz_active_particles is dependent"
	"on solkz benchmark.")
ENDIF ()

DEAL_II_INITIALIZE_CACHED_VARIABLES()

SET(TARGET "solkz_particles")
PROJECT(${TARGET})

INCLUDE_DIRECTORIES("${ASPECT_SOURCE_DIR}/benchmark/solkz")

ADD_LIBRARY(${TARGET} SHARED "${ASPECT_SOURCE_DIR}/benchmark/solkz/compositional_field/solkz_compositional_field.cc")

ASPECT_SETUP_PLUGIN(${TARGET})
